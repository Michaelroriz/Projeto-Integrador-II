CREATE TABLE CLIENTE
(
CLID integer PRIMARY KEY NOT NULL,
CLNOME VARCHAR(25),
CLCPF VARCHAR(11),
CLDTNASCIMENTO DATE(11),
CLSEXO VARCHAR(1),
CLENDERECO VARCHAR(25),
CLNUMERO INT(4),
CLCOMPLEMENTO VARCHAR(15),
CLBAIRRO VARCHAR(15),
CLCIDADE VARCHAR(18),
CLUF VARCHAR(2)
);

CREATE TABLE VEICULO
(
VID INTEGER PRIMARY KEY NOT NULL,
VPLACA VARCHAR(6) NOT NULL,
VMARCA VARCHAR(15) NOT NULL,
VMODELO VARCHAR(15) NOT NULL
);

CREATE TABLE FUNCIONARIO
(
FUNCID INTEGER PRIMARY KEY,
FNOME VARCHAR(25) NOT NULL,
FCPF VARCHAR(11) NOT NULL,
FSEXO VARCHAR(2) NOT NULL
);

CREATE TABLE AGENDAMENTO
(
AGID INTEGER PRIMARY KEY,
AGCLID INT,
AGVID INT,
AGFUNCID INT,
AGHRETIRADA DATETIME,
AGHDEVOLUCAO DATETIME,
FOREIGN KEY (AGCLID) REFERENCES CLIENTE(CLID),
FOREIGN KEY (AGVID) REFERENCES VEICULO(VID),
FOREIGN KEY (AGFUNCID) REFERENCES FUNCIONARIO(FUNCID)
);
